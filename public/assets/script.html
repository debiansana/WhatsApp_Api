Script WhatsApp BOT Monitoring Mikrotik

Script PPPoE :

On Up / Login
:local nama "$user";
:local ips [/ppp active get [find name=$nama] address];
:local up [/ppp active get [find name=$nama] uptime];
:local caller [/ppp active get [find name=$nama] caller-id];
:local service [/ppp active get [find name=$nama] service];
:local active [/ppp active print count];
:local profile [/ppp secret get [find name=$nama] profile];
:local datetime "Tanggal: $[/system clock get date] %0AJam: $[/system clock get time]";
:local lastdisc [/ppp secret get [find name=$user] last-disconnect-reason];
:local lastlogout [/ppp secret get [find name=$user] last-logged-out];
:local lastcall [/ppp secret get [find name=$user] last-caller-id];


/tool fetch http-header-field="content-type: application/x-www-form-urlencoded" http-method=post http-data="id_group=12xxxxxxxxxx@g.us&message=\E2\9C\85 TERHUBUNG%0A$datetime%0AUser : $user%0AIP Address : Private%0AMac Address : $caller%0ATipe Anggota : $profile%0AUptime : $up%0ATotal Active : $active Client%0AService : $service%0AKondisi Terputus : $lastdisc %0ATerakhir Terputus : $lastlogout %0ALast MAC : $lastcall" url="http://domain.com/send-group-message" keep-result=no;

On Down / Logout
:local lastdisc [/ppp secret get [find name=$user] last-disconnect-reason];
:local lastlogout [/ppp secret get [find name=$user] last-logged-out];
:local lastcall [/ppp secret get [find name=$user] last-caller-id];
:local active [/ppp active print count];
:local datetime "Tanggal: $[/system clock get date] %0AJam: $[/system clock get time]";


/tool fetch http-header-field="content-type: application/x-www-form-urlencoded" http-method=post http-data="id_group=12xxxxxxxxxxx@g.us&message=\E2\9D\8CTERPUTUS %0A$datetime%0AUser : $user%0AKondisi Terputus : $lastdisc %0ATerakhir Terputus : $lastlogout %0ALast MAC : $lastcall %0ATotal active: $active Client" url="http://domain.com/send-group-message" keep-result=no;


Netwatch - Monitoring Acess Point :

UP :
:local hh $host
:local datetime "Tanggal: $[/system clock get date] %0A Jam: $[/system clock get time]"
:local com [/tool netwatch get value-name=comment [find host=$hh] comment];

/tool fetch http-header-field="content-type: application/x-www-form-urlencoded" http-method=post http-data="id_group=12xxxxxxxxxx@g.us&message=$datetime %0A%0A$com %0AStatus : ON \E2\9C\85" url="http://domain.com/send-group-message" keep-result=no;

DOWN :
:local hh $host
:local datetime "Tanggal: $[/system clock get date] %0A Jam: $[/system clock get time]"
:local com [/tool netwatch get value-name=comment [find host=$hh] comment];

/tool fetch http-header-field="content-type: application/x-www-form-urlencoded" http-method=post http-data="id_group=12xxxxxxxxxx@g.us&message=$datetime %0A%0A$com %0AStatus : OFF \E2\9D\8C" url="http://domain.com/send-group-message" keep-result=no;
